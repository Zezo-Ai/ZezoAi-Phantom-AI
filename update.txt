        // ========== STATE MACHINE ==========
        class PhantomStateManager {
            constructor() {
                this.currentState = 'IDLE';
                this.stateHistory = [];
                this.taskData = null;
            }

            transition(toState) {
                const allowed = this.getAllowedTransitions(this.currentState);
                
                if (!allowed.includes(toState)) {
                    console.error(`Invalid transition: ${this.currentState} ‚Üí ${toState}`);
                    return false;
                }

                this.stateHistory.push({
                    from: this.currentState,
                    to: toState,
                    timestamp: Date.now(),
                    user: document.getElementById('displayUsername').textContent
                });

                this.currentState = toState;
                this.updateUI(toState);
                this.logTransition(toState);
                return true;
            }

            getAllowedTransitions(state) {
                const rules = {
                    'IDLE': ['COMPOSING'],
                    'COMPOSING': ['PROCESSING', 'IDLE'],
                    'PROCESSING': ['REVIEWING', 'COMPOSING'],
                    'REVIEWING': ['OUTPUTTING', 'COMPOSING'],
                    'OUTPUTTING': ['ESCALATING', 'COMPLETE', 'IDLE'],
                    'ESCALATING': ['OUTPUTTING', 'REVIEWING'],
                    'COMPLETE': ['IDLE']
                };
                return rules[state] || [];
            }

            updateUI(state) {
                document.getElementById('stateDisplay').textContent = state;
                
                const actionStatus = document.getElementById('actionStatus');
                if (state === 'REVIEWING') {
                    actionStatus.textContent = 'Pending Review';
                    actionStatus.className = 'badge processing';
                } else if (state === 'COMPLETE') {
                    actionStatus.textContent = 'Complete';
                    actionStatus.className = 'badge success';
                } else {
                    actionStatus.textContent = 'Awaiting';
                    actionStatus.className = 'badge';
                }
            }

            logTransition(state) {
                auditLogger.log('state_transition', {
                    from: this.stateHistory[this.stateHistory.length - 1]?.from || 'INIT',
                    to: state,
                    task_data: this.taskData
                });
            }

            getCurrentState() {
                return this.currentState;
            }

            getStateHistory() {
                return this.stateHistory;
            }
        }

        const stateManager = new PhantomStateManager();

        // ========== COST ESTIMATOR ==========
        class CostEstimator {
            constructor() {
                this.currentSessionCost = 0;
                this.totalCost = 0;
                this.taskHistory = [];
            }

            calculateBaseCost(taskData) {
                let cost = 0.01;
                
                if (taskData.prompt) {
                    const promptTokens = taskData.prompt.length / 4;
                    cost += promptTokens * 0.00001;
                }
                
                if (taskData.context) {
                    const contextTokens = taskData.context.length / 4;
                    cost += contextTokens * 0.00001;
                }
                
                const fileCount = taskData.files?.length || 0;
                cost += fileCount * 0.005;
                
                return cost;
            }

            calculateComplexityFactor(taskData) {
                let factor = 1.0;
                
                if (taskData.gating?.strictMode) factor *= 1.5;
                if (taskData.gating?.includeArtifacts) factor *= 1.2;
                if (taskData.gating?.includeLearning) factor *= 1.1;
                
                const promptLength = taskData.prompt?.length || 0;
                if (promptLength > 500) factor *= 1.5;
                else if (promptLength > 200) factor *= 1.2;
                
                return factor;
            }

            calculateEscalationProbability() {
                if (this.taskHistory.length < 3) return 0.1;
                
                const recent = this.taskHistory.slice(-10);
                const escalationRate = recent.filter(t => t.escalated).length / recent.length;
                return Math.min((0.1 + escalationRate) / 2, 0.9);
            }

            estimate(taskData) {
                const baseCost = this.calculateBaseCost(taskData);
                const tierMultiplier = parseInt(taskData.tier || 5);
                const complexity = this.calculateComplexityFactor(taskData);
                const escalationProb = this.calculateEscalationProbability();
                
                const estimatedCost = baseCost * tierMultiplier * complexity;
                const maxCost = estimatedCost * (1 + escalationProb * 20);
                
                return {
                    baseCost,
                    tierMultiplier,
                    complexity,
                    escalationProb,
                    estimatedCost,
                    maxCost
                };
            }

            addTaskCost(cost, escalated = false) {
                this.currentSessionCost += cost;
                this.totalCost += cost;
                this.taskHistory.push({ cost, escalated, timestamp: Date.now() });
                auditLogger.log('cost_incurred', { cost, escalated });
            }

            getSessionCost() {
                return this.currentSessionCost;
            }
        }
